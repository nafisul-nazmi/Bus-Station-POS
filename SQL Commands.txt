CREATE TABLE  DISTRICTS
   (	NAME VARCHAR2(4000) NOT NULL ENABLE, 
	 CONSTRAINT "DISTRICTS_PK" PRIMARY KEY (NAME) ENABLE
   );



CREATE TABLE  BUS 
   (	BUSID VARCHAR2(4000) NOT NULL ENABLE, 
	DAY VARCHAR2(4000) NOT NULL ENABLE, 
	TYPE VARCHAR2(4000) NOT NULL ENABLE, 
	FROM_D VARCHAR2(4000) NOT NULL ENABLE, 
	TO_D VARCHAR2(4000) NOT NULL ENABLE, 
	TIME VARCHAR2(4000) NOT NULL ENABLE, 
	FARE NUMBER NOT NULL ENABLE, 
	 CONSTRAINT "BUS_PK" PRIMARY KEY (BUSID) ENABLE, 
	 CONSTRAINT "BUS_FK" FOREIGN KEY (FROM_D)
	  REFERENCES  DISTRICTS (NAME) ENABLE, 
	 CONSTRAINT "BUS_FK2" FOREIGN KEY (TO_D)
	  REFERENCES  DISTRICTS (NAME) ENABLE
   );
